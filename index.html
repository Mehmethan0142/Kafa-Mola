from pathlib import Path import re src = Path("/mnt/data/index.html.txt") text = src.read_text(encoding="utf-8") # Üyelik ekranını HTML'den tamamen kaldır. text = re.sub( r'\s*
.*?
\s*(?=
', '
', 1) # Supabase sabitlerini kaldır. text = re.sub(r'const SUPABASE_URL=.*?;\s*const SUPABASE_KEY=.*?;\s*', '', text, count=1, flags=re.S) # auth ekranını ekran listesinden çıkar. text = text.replace('"auth","home"', '"home"') # Auth state değişkenlerini kaldır / sadeleştir. text = text.replace('let authType="login";\nlet currentUser=null;\nlet accessToken=null;\n', '') # authMode'dan enterApp öncesine kadar üyelik fonksiyonlarını kaldır. text = re.sub( r'function authMode\(mode\)\{.*?(?=function enterApp\(\))', '', text, count=1, flags=re.S ) # enterApp fonksiyonunu sadeleştir. text = re.sub( r'function enterApp\(\)\{\s*refresh\(\);\s*show\("home"\);\s*\}', 'function enterApp(){\nrefresh();\nshow("home");\n}', text, count=1, flags=re.S ) # show() içindeki currentUser koşulunu kaldır. text = text.replace( 'if(currentUser && navScreens.includes(id)){', 'if(navScreens.includes(id)){' ) # Profilde e-posta satırını kaldır. text = text.replace('

', '

Kafa Mola Yolculuğun

') text = re.sub(r'\s*profileEmail\.textContent=currentUser\?currentUser\.email:"";', '', text) # Çıkış butonunu değiştir. text = text.replace( '', '' ) # Supabase kaydet fonksiyonunu kaldır. text = re.sub( r'\nasync function kaydet\(veri\)\{.*?\n\}\n', '\n', text, count=1, flags=re.S ) # kaydet(...) çağrılarını kaldır. text = re.sub(r'\s*kaydet\(\{\s*durum:"checkin".*?\}\);\s*', '\n', text, count=1, flags=re.S) text = re.sub(r'\s*kaydet\(\{\s*durum:"deneyim".*?\}\);\s*', '\n', text, count=1, flags=re.S) # En alttaki session kontrolünü doğrudan uygulama başlangıcına çevir. text = re.sub( r'if\(loadSession\(\)\)\{\s*enterApp\(\);\s*\}else\{\s*show\("auth"\);\s*\}', 'enterApp();', text, count=1, flags=re.S ) out = Path("/mnt/data/index.html") out.write_text(text, encoding="utf-8") print("Hazır: index.html")
